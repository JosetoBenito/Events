private void Navigate(Type view)
{
	var type = view.Name;

	switch (type)
	{
		case "HomeView":
			SplitViewFrame.Navigate(view, _driverStanding);
			break;
		case "StandingsView":
			SplitViewFrame.Navigate(view, _driverStanding);
			break;
		case "AboutView":
			SplitViewFrame.Navigate(view);
			break;
	}
}
